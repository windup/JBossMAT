132        // Classfile
133        public void visitClassfile(Classfile classfile) {
134            String className = classfile.getClassName();
135            Logger.getLogger(getClass()).debug("VisitClassfile():");
136            Logger.getLogger(getClass()).debug("    class = \"" + className + "\"");
137    
138            fireBeginClass(classfile);
139            
140            setCurrentMethod(null);
141            setCurrentClass(getMetricsFactory().createClassMetrics(className));
142            setCurrentGroup(getCurrentClass().getParent());
143            setCurrentProject(getCurrentGroup().getParent());
144    
145            getMetricsFactory().includeClassMetrics(getCurrentClass());
146    
147            getCurrentProject().addToMeasurement(BasicMeasurements.PACKAGES, getCurrentGroup().getName());
148    
149            if (classfile.isPublic()) {
150                getCurrentProject().addToMeasurement(BasicMeasurements.PUBLIC_CLASSES, className);
151                getCurrentGroup().addToMeasurement(BasicMeasurements.PUBLIC_CLASSES, className);
152            } else {
153                getCurrentProject().addToMeasurement(BasicMeasurements.PACKAGE_CLASSES, className);
154                getCurrentGroup().addToMeasurement(BasicMeasurements.PACKAGE_CLASSES, className);
155            }
156    
157            if (classfile.isFinal()) {
158                getCurrentProject().addToMeasurement(BasicMeasurements.FINAL_CLASSES, className);
159                getCurrentGroup().addToMeasurement(BasicMeasurements.FINAL_CLASSES, className);
160            }
161    
162            if (classfile.isSuper()) {
163                getCurrentProject().addToMeasurement(BasicMeasurements.SUPER_CLASSES, className);
164                getCurrentGroup().addToMeasurement(BasicMeasurements.SUPER_CLASSES, className);
165            }
166    
167            if (classfile.isInterface()) {
168                getCurrentProject().addToMeasurement(BasicMeasurements.INTERFACES, className);
169                getCurrentGroup().addToMeasurement(BasicMeasurements.INTERFACES, className);
170            }
171    
172            if (classfile.isAbstract()) {
173                getCurrentProject().addToMeasurement(BasicMeasurements.ABSTRACT_CLASSES, className);
174                getCurrentGroup().addToMeasurement(BasicMeasurements.ABSTRACT_CLASSES, className);
175            }
176    
177            if (classfile.getSuperclassIndex() != 0) {
178                classfile.getRawSuperclass().accept(this);
179    
180                getMetricsFactory().createClassMetrics(classfile.getSuperclassName()).addToMeasurement(BasicMeasurements.SUBCLASSES);
181    
182                Classfile superclass = classfile.getLoader().getClassfile(classfile.getSuperclassName());
183                if (superclass != null) {
184                    getCurrentClass().addToMeasurement(BasicMeasurements.DEPTH_OF_INHERITANCE, computeDepthOfInheritance(superclass));
185                }
186            }
187    
188            for (Class_info class_info : classfile.getAllInterfaces()) {
189                class_info.accept(this);
190            }
191    
192            for (Field_info field : classfile.getAllFields()) {
193                field.accept(this);
194            }
195    
196            for (Method_info method : classfile.getAllMethods()) {
197                method.accept(this);
198            }
199    
200            sloc = 1;
201            isSynthetic = false;
202    
203            for (Attribute_info attribute : classfile.getAttributes()) {
204                attribute.accept(this);
205            }
206            
207            if (!isSynthetic) {
208                getCurrentClass().addToMeasurement(BasicMeasurements.CLASS_SLOC, sloc);
209            }
210    
211            fireEndClass(classfile, getCurrentClass());
212        }

216            Logger.getLogger(getClass()).debug("VisitClass_info():");
217            Logger.getLogger(getClass()).debug("    name = \"" + entry.getName() + "\"");
218            if (entry.getName().startsWith("[")) {
219                addClassDependencies(processDescriptor(entry.getName()));
220            } else {
221                addClassDependency(entry.getName());
222            }

254        public void visitField_info(Field_info entry) {
255            String fullName = entry.getFullName();
256            getCurrentClass().addToMeasurement(BasicMeasurements.ATTRIBUTES, fullName);
257    
258            Logger.getLogger(getClass()).debug("VisitField_info(" + entry.getFullSignature() + ")");
259            Logger.getLogger(getClass()).debug("Current class: " + getCurrentClass().getName());
260            Logger.getLogger(getClass()).debug("Access flag: " + entry.getAccessFlag());
261            Logger.getLogger(getClass()).debug("Public: " + entry.isPublic());
262            Logger.getLogger(getClass()).debug("Private: " + entry.isPrivate());
263            Logger.getLogger(getClass()).debug("Protected: " + entry.isProtected());
264            Logger.getLogger(getClass()).debug("Static: " + entry.isStatic());
265            
266            if (entry.isPublic()) {
267                getCurrentClass().addToMeasurement(BasicMeasurements.PUBLIC_ATTRIBUTES, fullName);
268            } else if (entry.isPrivate()) {
269                getCurrentClass().addToMeasurement(BasicMeasurements.PRIVATE_ATTRIBUTES, fullName);
270            } else if (entry.isProtected()) {
271                getCurrentClass().addToMeasurement(BasicMeasurements.PROTECTED_ATTRIBUTES, fullName);
272            } else {
273                getCurrentClass().addToMeasurement(BasicMeasurements.PACKAGE_ATTRIBUTES, fullName);
274            }
275    
276            if (entry.isStatic()) {
277                getCurrentClass().addToMeasurement(BasicMeasurements.STATIC_ATTRIBUTES, fullName);
278            }
279    
280            if (entry.isFinal()) {
281                getCurrentClass().addToMeasurement(BasicMeasurements.FINAL_ATTRIBUTES, fullName);
282            }
283    
284            if (entry.isVolatile()) {
285                getCurrentClass().addToMeasurement(BasicMeasurements.VOLATILE_ATTRIBUTES, fullName);
286            }
287    
288            if (entry.isTransient()) {
289                getCurrentClass().addToMeasurement(BasicMeasurements.TRANSIENT_ATTRIBUTES, fullName);
290            }
291    
292            sloc = 1;
293            isSynthetic = entry.isSynthetic();
294            
295            super.visitField_info(entry);
296            
297            if (!isSynthetic) {
298                getCurrentClass().addToMeasurement(BasicMeasurements.CLASS_SLOC, sloc);
299            }
300    
301            addClassDependencies(processDescriptor(entry.getDescriptor()));
302        }

304        public void visitMethod_info(Method_info entry) {
305            fireBeginMethod(entry);
306    
307            String fullSignature = entry.getFullSignature();
308            setCurrentMethod(getMetricsFactory().createMethodMetrics(fullSignature));
309            getMetricsFactory().includeMethodMetrics(getCurrentMethod());
310            
311            Logger.getLogger(getClass()).debug("VisitMethod_info(" + fullSignature + ")");
312            Logger.getLogger(getClass()).debug("Current class: " + getCurrentClass().getName());
313            Logger.getLogger(getClass()).debug("Access flag: " + entry.getAccessFlag());
314            Logger.getLogger(getClass()).debug("Public: " + entry.isPublic());
315            Logger.getLogger(getClass()).debug("Private: " + entry.isPrivate());
316            Logger.getLogger(getClass()).debug("Protected: " + entry.isProtected());
317            Logger.getLogger(getClass()).debug("Static: " + entry.isStatic());
318    
319            sloc = 0;
320            isSynthetic = entry.isSynthetic();
321            
322            if (entry.isPublic()) {
323                getCurrentClass().addToMeasurement(BasicMeasurements.PUBLIC_METHODS, fullSignature);
324            } else if (entry.isPrivate()) {
325                getCurrentClass().addToMeasurement(BasicMeasurements.PRIVATE_METHODS, fullSignature);
326            } else if (entry.isProtected()) {
327                getCurrentClass().addToMeasurement(BasicMeasurements.PROTECTED_METHODS, fullSignature);
328            } else {
329                getCurrentClass().addToMeasurement(BasicMeasurements.PACKAGE_METHODS, fullSignature);
330            }
331    
332            if (entry.isStatic()) {
333                getCurrentClass().addToMeasurement(BasicMeasurements.STATIC_METHODS, fullSignature);
334            }
335    
336            if (entry.isFinal()) {
337                getCurrentClass().addToMeasurement(BasicMeasurements.FINAL_METHODS, fullSignature);
338            }
339    
340            if (entry.isSynchronized()) {
341                getCurrentClass().addToMeasurement(BasicMeasurements.SYNCHRONIZED_METHODS, fullSignature);
342            }
343    
344            if (entry.isNative()) {
345                getCurrentClass().addToMeasurement(BasicMeasurements.NATIVE_METHODS, fullSignature);
346            }
347    
348            if (entry.isAbstract()) {
349                getCurrentClass().addToMeasurement(BasicMeasurements.ABSTRACT_METHODS, fullSignature);
350                sloc = 1;
351            }
352    
353            getCurrentMethod().addToMeasurement(BasicMeasurements.PARAMETERS, DescriptorHelper.getParameterCount(entry.getDescriptor()));
354            
355            super.visitMethod_info(entry);
356            
357            if (!isSynthetic) {
358                getCurrentMethod().addToMeasurement(BasicMeasurements.SLOC, sloc);
359            }
360    
361            addClassDependencies(processDescriptor(entry.getDescriptor()));
362    
363            fireEndMethod(entry, getCurrentMethod());
364        }

370        public void visitSynthetic_attribute(Synthetic_attribute attribute) {
371            Object owner = attribute.getOwner();
372    
373            isSynthetic = true;
374            
375            if (owner instanceof Classfile) {
376                String className = ((Classfile) owner).getClassName();
377                getCurrentProject().addToMeasurement(BasicMeasurements.SYNTHETIC_CLASSES, className);
378                getCurrentGroup().addToMeasurement(BasicMeasurements.SYNTHETIC_CLASSES, className);
379            } else if (owner instanceof Field_info) {
380                getCurrentClass().addToMeasurement(BasicMeasurements.SYNTHETIC_ATTRIBUTES, ((Field_info) owner).getFullName());
381            } else if (owner instanceof Method_info) {
382                getCurrentClass().addToMeasurement(BasicMeasurements.SYNTHETIC_METHODS, ((Method_info) owner).getFullSignature());
383            } else {
384                Logger.getLogger(getClass()).warn("Synthetic attribute on unknown Visitable: " + owner.getClass().getName());
385            }
386        }

408        public void visitInstruction(Instruction helper) {
409            super.visitInstruction(helper);
410    
411            /*
412             *  We can skip the "new" (0xbb) instruction as it is always
413             *  followed by a call to the constructor method.
414             */
415    
416            switch (helper.getOpcode()) {
417                case 0x12: // ldc
418                case 0x13: // ldc_w
419                case 0xb2: // getstatic
420                case 0xb3: // putstatic
421                case 0xb4: // getfield
422                case 0xb5: // putfield
423                case 0xb6: // invokevirtual
424                case 0xb7: // invokespecial
425                case 0xb8: // invokestatic
426                case 0xb9: // invokeinterface
427                // case 0xbb: // new
428                case 0xbd: // anewarray
429                case 0xc0: // checkcast
430                case 0xc1: // instanceof
431                case 0xc5: // multianewarray
432                    helper.getIndexedConstantPoolEntry().accept(this);
433                    break;
434                default:
435                    // Do nothing
436                    break;
437            }
438        }

493        boolean isInnerClassOfCurrentClass(InnerClass helper) {
494            boolean result;
495    
496            if (helper.getOuterClassInfo().equals("")) {
497                result = perl.match("/^" + helper.getInnerClasses().getClassfile().getClassName() + "\\$\\d+$/", helper.getInnerClassInfo());
498            } else {
499                result = helper.getOuterClassInfo().equals(helper.getInnerClasses().getClassfile().getClassName());
500            }
501    
502            return result;
503        }

515        private int computeDepthOfInheritance(Classfile classfile) {
516            int result = 1;
517            
518            if (classfile != null && classfile.getSuperclassIndex() != 0) {
519                Classfile superclass = classfile.getLoader().getClassfile(classfile.getSuperclassName());
520                result += computeDepthOfInheritance(superclass);
521            }
522    
523            return result;
524        }

526        private Collection<String> processDescriptor(String str) {
527            Collection<String> result = new LinkedList<String>();
528            
529            Logger.getLogger(getClass()).debug("ProcessDescriptor: " + str);
530    
531            int currentPos = 0;
532            int startPos;
533            int endPos;
534    
535            while ((startPos = str.indexOf('L', currentPos)) != -1) {
536                if ((endPos = str.indexOf(';', startPos)) != -1) {
537                    String classname = ClassNameHelper.path2ClassName(str.substring(startPos + 1, endPos));
538                    result.add(classname);
539                    currentPos = endPos + 1;
540                } else {
541                    currentPos = startPos + 1;
542                }
543            }
544    
545            Logger.getLogger(getClass()).debug("ProcessDescriptor: " + result);
546            
547            return result;
548        }

556        private void addClassDependency(String name) {
557            Logger.getLogger(getClass()).debug("AddClassDependency(\"" + name + "\") ...");
558    
559            if (!getCurrentClass().getName().equals(name) && isInFilter(name)) {
560                Metrics other = getMetricsFactory().createClassMetrics(name);
561                    
562                if (getCurrentMethod() != null && isInScope(getCurrentMethod().getName())) {
563                    Logger.getLogger(getClass()).debug("AddClassDependency " + getCurrentMethod().getName() + " -> " + name + " ...");
564                    
565                    if (getCurrentClass().getParent().equals(other.getParent())) {
566                        Logger.getLogger(getClass()).debug("Intra-Package ...");
567                        getCurrentMethod().addToMeasurement(BasicMeasurements.OUTBOUND_INTRA_PACKAGE_CLASS_DEPENDENCIES, other.getName());
568                        other.addToMeasurement(BasicMeasurements.INBOUND_INTRA_PACKAGE_METHOD_DEPENDENCIES, getCurrentMethod().getName());
569                    } else {
570                        Logger.getLogger(getClass()).debug("Extra-Package ...");
571                        getCurrentMethod().addToMeasurement(BasicMeasurements.OUTBOUND_EXTRA_PACKAGE_CLASS_DEPENDENCIES, other.getName());
572                        other.addToMeasurement(BasicMeasurements.INBOUND_EXTRA_PACKAGE_METHOD_DEPENDENCIES, getCurrentMethod().getName());
573                    }
574                } else if (isInScope(getCurrentClass().getName())) {
575                    Logger.getLogger(getClass()).debug("AddClassDependency " + getCurrentClass().getName() + " -> " + name + " ...");
576                    
577                    if (getCurrentClass().getParent().equals(other.getParent())) {
578                        Logger.getLogger(getClass()).debug("Intra-Package ...");
579                        getCurrentClass().addToMeasurement(BasicMeasurements.OUTBOUND_INTRA_PACKAGE_DEPENDENCIES, other.getName());
580                        other.addToMeasurement(BasicMeasurements.INBOUND_INTRA_PACKAGE_DEPENDENCIES, getCurrentClass().getName());
581                    } else {
582                        Logger.getLogger(getClass()).debug("Extra-Package ...");
583                        getCurrentClass().addToMeasurement(BasicMeasurements.OUTBOUND_EXTRA_PACKAGE_DEPENDENCIES, other.getName());
584                        other.addToMeasurement(BasicMeasurements.INBOUND_EXTRA_PACKAGE_DEPENDENCIES, getCurrentClass().getName());
585                    }
586                }
587            }
588        }

590        private void addMethodDependency(String name) {
591            Logger.getLogger(getClass()).debug("AddMethodDependency " + getCurrentMethod().getName() + " -> " + name + " ...");
592    
593            if (!getCurrentMethod().getName().equals(name) && isInScope(getCurrentMethod().getName()) && isInFilter(name)) {
594                Metrics other = getMetricsFactory().createMethodMetrics(name);
595                
596                if (getCurrentClass().equals(other.getParent())) {
597                    Logger.getLogger(getClass()).debug("Intra-Class ...");
598                    getCurrentMethod().addToMeasurement(BasicMeasurements.OUTBOUND_INTRA_CLASS_FEATURE_DEPENDENCIES, other.getName());
599                    other.addToMeasurement(BasicMeasurements.INBOUND_INTRA_CLASS_METHOD_DEPENDENCIES, getCurrentMethod().getName());
600                } else if (getCurrentGroup().equals(other.getParent().getParent())) {
601                    Logger.getLogger(getClass()).debug("Intra-Package ...");
602                    getCurrentMethod().addToMeasurement(BasicMeasurements.OUTBOUND_INTRA_PACKAGE_FEATURE_DEPENDENCIES, other.getName());
603                    other.addToMeasurement(BasicMeasurements.INBOUND_INTRA_PACKAGE_METHOD_DEPENDENCIES, getCurrentMethod().getName());
604                } else {
605                    Logger.getLogger(getClass()).debug("Extra-Package ...");
606                    getCurrentMethod().addToMeasurement(BasicMeasurements.OUTBOUND_EXTRA_PACKAGE_FEATURE_DEPENDENCIES, other.getName());
607                    other.addToMeasurement(BasicMeasurements.INBOUND_EXTRA_PACKAGE_METHOD_DEPENDENCIES, getCurrentMethod().getName());
608                }
609            }
610        }
