001    package com.jeantessier.classreader.impl;
002    
003    import java.io.*;
004    
005    import org.apache.log4j.*;
006    
007    import com.jeantessier.classreader.*;
008    
009    public class LocalVariableType implements com.jeantessier.classreader.LocalVariableType {
010        private LocalVariableTypeTable_attribute localVariableTypeTable;
011        private int startPC;
012        private int length;
013        private int nameIndex;
014        private int signatureIndex;
015        private int index;
016    
017        public LocalVariableType(LocalVariableTypeTable_attribute localVariableTypeTable, DataInputStream in) throws IOException {
018            this.localVariableTypeTable = localVariableTypeTable;
019    
020            startPC = in.readUnsignedShort();
021            Logger.getLogger(getClass()).debug("start PC: " + startPC);
022    
023            length = in.readUnsignedShort();
024            Logger.getLogger(getClass()).debug("length: " + length);
025    
026            nameIndex = in.readUnsignedShort();
027            Logger.getLogger(getClass()).debug("name: " + nameIndex + " (" + getName() + ")");
028    
029            signatureIndex = in.readUnsignedShort();
030            Logger.getLogger(getClass()).debug("signature: " + signatureIndex + " (" + getSignature() + ")");
031    
032            index = in.readUnsignedShort();
033            Logger.getLogger(getClass()).debug("index: " + index);
034        }
035    
036        public LocalVariableTypeTable_attribute getLocalVariableTypeTable() {
037            return localVariableTypeTable;
038        }
039    
040        public int getStartPC() {
041            return startPC;
042        }
043    
044        public int getLength() {
045            return length;
046        }
047    
048        public int getNameIndex() {
049            return nameIndex;
050        }
051    
052        public UTF8_info getRawName() {
053            return (UTF8_info) getLocalVariableTypeTable().getClassfile().getConstantPool().get(getNameIndex());
054        }
055    
056        public String getName() {
057            return getRawName().toString();
058        }
059    
060        public int getSignatureIndex() {
061            return signatureIndex;
062        }
063    
064        public UTF8_info getRawSignature() {
065            return (UTF8_info) getLocalVariableTypeTable().getClassfile().getConstantPool().get(getSignatureIndex());
066        }
067    
068        public String getSignature() {
069            return getRawSignature().toString();
070        }
071    
072        public int getIndex() {
073            return index;
074        }
075    
076        public String toString() {
077            return "Local variable type " + getSignature() + " " + getName();
078        }
079    
080        public void accept(Visitor visitor) {
081            visitor.visitLocalVariableType(this);
082        }
083    }
